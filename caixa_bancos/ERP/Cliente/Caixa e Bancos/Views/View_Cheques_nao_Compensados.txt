CREATE VIEW VIEW_FIN_CHEQUE_NAO_COMPENSADO(
	ID,
    ID_CONTA_CAIXA,
    NOME_CONTA_CAIXA,
    TALAO,
	NUMERO_TALAO,
    NUMERO_CHEQUE,
    STATUS_CHEQUE,
    DATA_STATUS,
	VALOR)
AS
select
  0 as ID,
  CC.ID AS ID_CONTA_CAIXA, CC.NOME AS NOME_CONTA_CAIXA, 
  T.TALAO, T.NUMERO AS NUMERO_TALAO, 
  C.NUMERO AS NUMERO_CHEQUE, C.STATUS_CHEQUE, C.DATA_STATUS, VALOR
 
from 
  CHEQUE C 
	inner join TALONARIO_CHEQUE T on (C.ID_TALONARIO_CHEQUE = T.ID)
    inner join CONTA_CAIXA CC on (T.ID_CONTA_CAIXA = CC.ID)
	inner join FIN_CHEQUE_EMITIDO CE on (CE.ID_CHEQUE = C.ID)

where
  C.STATUS_CHEQUE <> 'C'
