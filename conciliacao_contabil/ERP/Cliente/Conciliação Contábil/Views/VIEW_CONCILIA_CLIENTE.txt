CREATE
VIEW t2tierp.VIEW_CONCILIA_CLIENTE
AS

select
	PR.ID,
	PR.ID_FIN_PARCELA_RECEBER,
	PR.ID_FIN_TIPO_RECEBIMENTO, 
	PR.ID_CONTA_CAIXA,
	PR.DATA_RECEBIMENTO,
	PR.TAXA_JURO,
	PR.TAXA_MULTA,
	PR.TAXA_DESCONTO,
	PR.VALOR_JURO,
	PR.VALOR_MULTA,
	PR.VALOR_DESCONTO,
	PR.VALOR_RECEBIDO,
	PR.HISTORICO,
	LR.ID_CLIENTE
from
	FIN_LANCAMENTO_RECEBER LR
	INNER JOIN FIN_PARCELA_RECEBER PPR ON (PPR.ID_FIN_LANCAMENTO_RECEBER = LR.ID)
	INNER JOIN FIN_PARCELA_RECEBIMENTO PR ON (PR.ID_FIN_PARCELA_RECEBER = PPR.ID)
