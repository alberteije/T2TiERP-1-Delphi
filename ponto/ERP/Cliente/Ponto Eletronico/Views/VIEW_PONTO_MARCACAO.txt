CREATE
VIEW t2tierp.VIEW_PONTO_MARCACAO
AS
(
select 
	PM.*, C.PIS_NUMERO, P.NOME AS PESSOA_NOME, PR.NUMERO_SERIE
from
	PONTO_MARCACAO PM
	INNER JOIN COLABORADOR C ON (PM.ID_COLABORADOR = C.ID)
	INNER JOIN PESSOA P ON (C.ID_PESSOA = P.ID)
	INNER JOIN PONTO_RELOGIO PR ON (PM.ID_PONTO_RELOGIO = PR.ID)
)
