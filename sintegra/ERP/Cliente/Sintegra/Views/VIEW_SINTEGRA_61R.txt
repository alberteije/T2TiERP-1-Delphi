	CREATE
	VIEW VIEW_SINTEGRA_61R
	AS
	(
	select
		P.GTIN, 
		NF2C.DATA_EMISSAO AS DATA_EMISSAO, 
		MONTH(NF2C.DATA_EMISSAO) AS MES_EMISSAO, 
		YEAR(NF2C.DATA_EMISSAO) AS ANO_EMISSAO, 
		NF2D.ECF_ICMS_ST, 
		SUM(NF2D.QUANTIDADE) AS SOMA_QUANTIDADE, 
		SUM(NF2D.VALOR_TOTAL) AS SOMA_ITEM, 
		SUM(NF2D.BASE_ICMS) AS SOMA_BASE_ICMS 
	from 
		ECF_NOTA_FISCAL_CABECALHO NF2C, ECF_NOTA_FISCAL_DETALHE NF2D, PRODUTO P
	where 
		NF2D.ID_PRODUTO=P.ID and 
		NF2D.ID_NF_CABECALHO = NF2C.ID
	group by 
		MES_EMISSAO, ANO_EMISSAO, P.GTIN
	)
