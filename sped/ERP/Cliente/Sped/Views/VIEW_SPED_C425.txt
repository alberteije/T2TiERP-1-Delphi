	CREATE
	VIEW t2tierp.VIEW_SPED_C425
	AS
	(
	select 
		VD.ID_ECF_PRODUTO, 
		U.SIGLA AS DESCRICAO_UNIDADE, 
		VD.TOTALIZADOR_PARCIAL,
		VC.DATA_VENDA,
		SUM(VD.QUANTIDADE) AS SOMA_QUANTIDADE, 
		SUM(VD.VALOR_TOTAL) AS SOMA_ITEM, 
		SUM(VD.PIS) AS SOMA_PIS, 
		SUM(VD.COFINS) AS SOMA_COFINS 
	from 
		ECF_VENDA_DETALHE VD, ECF_VENDA_CABECALHO VC, PRODUTO P, UNIDADE_PRODUTO U 
	where 
		VD.ID_ECF_VENDA_CABECALHO = VC.ID and 
		VD.ID_ECF_PRODUTO = P.ID and 
		P.ID_UNIDADE_PRODUTO = U.ID
	group by 
		ID_ECF_PRODUTO, U.SIGLA, ID_UNIDADE_PRODUTO, TOTALIZADOR_PARCIAL,DATA_VENDA
	)
