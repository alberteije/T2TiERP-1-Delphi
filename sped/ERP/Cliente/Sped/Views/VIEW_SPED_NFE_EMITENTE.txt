CREATE
VIEW t2tierp.VIEW_SPED_NFE_EMITENTE
AS

select 
	DISTINCT(NFE.ID_FORNECEDOR) AS ID,
	E.RAZAO_SOCIAL,
	E.CPF_CNPJ,
	E.INSCRICAO_ESTADUAL,
	E.CODIGO_MUNICIPIO,
	E.SUFRAMA,
	E.LOGRADOURO,
	E.NUMERO,
	E.COMPLEMENTO,
	E.BAIRRO
from
	NFE_CABECALHO NFE 
	INNER JOIN NFE_EMITENTE E ON (E.ID_NFE_CABECALHO = NFE.ID)
where 
	NFE.ID_FORNECEDOR IS NOT NULL
